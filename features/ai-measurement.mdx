## Overview

LogiScan's AI measurement feature uses computer vision and optical character recognition (OCR) to automatically extract package dimensions and weight from shipping label photos. This eliminates manual data entry and reduces measurement errors in warehouse operations.

## How It Works

<Steps>
  <Step title="Capture Label Photo">

## Implementation

### Measurement Capture

The measurement capture flow in the scan details page:

```dart
Future<void> _handleMeasurementCapture() async {
  if (_isOverMaxImages()) return;

  final XFile? photo = await _pickMeasurementPhoto();
  if (photo == null) return;

  setState(() {
    _isProcessingMeasurements = true;
    _images.add(photo);
  });

  try {
    final measurementService = context.read<MeasurementService>();
    final request = await _buildProcessMeasurementRequest(photo);
    if (request == null) {
      setState(() => _isProcessingMeasurements = false);
      return;
    }

    final response = await measurementService.processMeasurementData(request);
    if (!mounted) return;

    if (response.isSuccessful) {
      _applyMeasurementResponse(response.content);
    } else {
      final msg = response.messageDetail;
      if (mounted && msg != null && msg.isNotEmpty) {
        MessageHelper.showIconSnackBar(
          context,
          message: msg,
          isSuccess: false,
        );
      }
    }
  } catch (_) {
    if (!mounted) return;
    MessageHelper.showIconSnackBar(
      context,
      message: 'Unexpected error capturing measurements.',
      isSuccess: false,
    );
  } finally {
    if (mounted) {
      setState(() => _isProcessingMeasurements = false);
    }
  }
}
```

### Request Building

Prepare the measurement request with encoded image:

```dart
Future<ProcessMeasurementDataRequest?> _buildProcessMeasurementRequest(
  XFile photo,
) async {
  final tracking = _codeController.text.trim();
  if (tracking.isEmpty) {
    MessageHelper.showIconSnackBar(
      context,
      message:
          'Enter or scan a tracking number before capturing measurements.',
      isSuccess: false,
    );
    return null;
  }

  final bytes = await photo.readAsBytes();
  final encoded = base64Encode(bytes);
  final fileName = photo.name;
  final ext = fileName.contains('.') ? fileName.split('.').last : 'jpg';

  return ProcessMeasurementDataRequest(
    file: RequestEncodeFile(
      encodeContent: encoded,
      fileExtension: ext,
      fileName: fileName,
    ),
  );
}
```

### Response Processing

Apply extracted measurements to form fields:

```dart
void _applyMeasurementResponse(ProcessMeasurementDataResponse? data) {
  if (data == null) {
    return;
  }

  setState(() {
    if (data.length != null) {
      _lengthController.text = _formatNumber(data.length!);
    }
    if (data.width != null) {
      _widthController.text = _formatNumber(data.width!);
    }
    if (data.height != null) {
      _heightController.text = _formatNumber(data.height!);
    }
    if (data.weight != null) {
      _weightController.text = _formatNumber(data.weight!);
    }
  });
}

String _formatNumber(double value) =>
    value.toStringAsFixed(value.truncateToDouble() == value ? 0 : 2);
```

## MeasurementService

The service handles API communication:

```dart
class MeasurementService {
  final HttpService _http;

  MeasurementService(this._http);

  Future<ApiResponse<ProcessMeasurementDataResponse>>
      processMeasurementData(ProcessMeasurementDataRequest request) async {
    try {
      final response = await _http.post<ProcessMeasurementDataResponse>(
        ApiEndpoints.processMeasurement,
        request.toJson(),
        (json) => ProcessMeasurementDataResponse.fromJson(json),
      );

      if (!response.isSuccessful) {
        return ApiResponse.error(
          messageDetail: response.messageDetail,
          content: ProcessMeasurementDataResponse.empty(),
        );
      }

      return response;
    } catch (_) {
      return ApiResponse.error(
        messageDetail: null,
        content: ProcessMeasurementDataResponse.empty(),
      );
    }
  }
}
```

### ProcessMeasurementDataResponse

Response model with extracted measurements:

```dart
class ProcessMeasurementDataResponse {
  final double? weight;
  final double? width;
  final double? length;
  final double? height;

  const ProcessMeasurementDataResponse({
    this.weight,
    this.width,
    this.length,
    this.height,
  });

  factory ProcessMeasurementDataResponse.empty() =>
      const ProcessMeasurementDataResponse();

  factory ProcessMeasurementDataResponse.fromJson(Map<String, dynamic> json) {
    final content = json['content'] as Map<String, dynamic>? ?? json;

    double? asDouble(dynamic value) {
      if (value == null) return null;
      if (value is num) return value.toDouble();
      return double.tryParse(value.toString());
    }

    return ProcessMeasurementDataResponse(
      weight: asDouble(content['weight']),
      width: asDouble(content['width']),
      length: asDouble(content['length']),
      height: asDouble(content['height']),
    );
  }
}
```

**Endpoint**: `POST /api/mobile/v1.0/ProcessPackage/process-measurement-data`

**Request**:
```json

  "file": {

    "fileName": "label_photo.jpg"
```

**Response**:
```json
{
  "code": 1,
  "message": "Success",
  "content": {
    "weight": 15.5,
    "width": 12,
    "length": 10,
    "height": 8

}
```

<Note>

## Best Practices

### Photo Capture Tips

<AccordionGroup>
  <Accordion title="Lighting Conditions">
    - Use adequate lighting without harsh shadows
    - Avoid glare on glossy labels
    - Natural or bright white LED lighting works best
    - Position light source from the side, not directly above
  </Accordion>
  
  <Accordion title="Label Positioning">
    - Capture the entire label in frame
    - Hold device parallel to label surface
    - Ensure label is flat and not wrinkled
    - Minimum distance: 6-12 inches from label
  </Accordion>
  
  <Accordion title="Image Quality">
    - Use device's auto-focus feature
    - Ensure text is sharp and readable
    - Avoid motion blur by holding device steady
    - Clean camera lens before capture
  </Accordion>
</AccordionGroup>

## Measurement Units

### Dimensions

- **Unit**: Inches (in)
- **Format**: Integer values (rounded)
- **Typical Range**: 1-60 inches per dimension
- **Validation**: Must be positive integers

### Weight

- **Unit**: Pounds (lb)
- **Format**: Decimal with up to 2 places
- **Typical Range**: 0.1-150 pounds
- **Validation**: Must be positive number

```dart
class _NumberField extends StatelessWidget {
  final TextEditingController controller;
  final String label;
  final bool integerOnly;
  final int? maxDecimals;

  @override
  Widget build(BuildContext context) {
    final List<TextInputFormatter> formatters;

    if (integerOnly) {
      formatters = [FilteringTextInputFormatter.digitsOnly];
    } else {
      final decimals = maxDecimals ?? 2;
      final pattern = RegExp('^\\d*\\.?\\d{0,$decimals} ?');
      formatters = [FilteringTextInputFormatter.allow(pattern)];
    }

    return TextField(
      controller: controller,
      keyboardType: integerOnly
          ? TextInputType.number
          : const TextInputType.numberWithOptions(decimal: true),
      inputFormatters: formatters,
      decoration: InputDecoration(
        isDense: true,
        filled: true,
        fillColor: Colors.white,
        labelText: label,
        border: const OutlineInputBorder(
          borderRadius: BorderRadius.all(Radius.circular(12)),
        ),
      ),
    );
  }
}
```

### Owner Detection Model

```dart
enum OwnerType {
  globalShipping,
  gbi,
  unknown,
}

class GetPackageOwnerByImageResponse {
  final OwnerType ownerPackage;
  final String? color;

  const GetPackageOwnerByImageResponse({
    required this.ownerPackage,
    required this.color,
  });

  factory GetPackageOwnerByImageResponse.fromJson(Map<String, dynamic> json) {
    final content = json['content'] as Map<String, dynamic>? ?? json;
    final ownerRaw = (content['ownerPackage'] as String?) ?? 'Unknown';
    final normalized = ownerRaw.toLowerCase();

    OwnerType owner;
    switch (normalized) {
      case 'glovalshipping':
      case 'globalshipping':
        owner = OwnerType.globalShipping;
        break;
      case 'gbi':
        owner = OwnerType.gbi;
        break;
      default:
        owner = OwnerType.unknown;
        break;
    }

    return GetPackageOwnerByImageResponse(
      ownerPackage: owner,
      color: content['color'] as String?,
    );
  }
}
```
