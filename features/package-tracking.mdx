## Overview

LogiScan's package tracking system provides comprehensive visibility into package status, history, and details. Users can query packages by tracking number or package ID, view detailed information including dimensions and weight, access tracking history, and manage package photos.

```dart
Future<ApiResponse<GetPackageTrackingDetailsResponse>>
    getPackageTrackingDetails(

  try {

    );
  }
}
```

### Tracking Details UI

The main tracking interface with auto-load on scan:

```dart
Future<void> _loadDetailsAndImages() async {
  final tracking = _trackingController.text.trim();

  if (tracking.isEmpty) {
    MessageHelper.showIconSnackBar(
      context,
      message: 'Enter a tracking number.',
      isSuccess: false,
    );
    setState(() {
      _details = null;
      _photos.clear();
      _pendingDeleteStoredFileIds.clear();
    });
    _trackingFocusNode.requestFocus();
    return;
  }

  try {
    final measurementService = context.read<MeasurementService>();

    final response = await measurementService.getPackageTrackingDetails(
      tracking,
    );

    if (!mounted) return;

    final details = response.content;
    if (response.isSuccessful &&
        details != null &&
        details.items.isNotEmpty) {
      final List<_PhotoItem> loadedPhotos = details.images
          .map<_PhotoItem>((img) => _PhotoItem.remote(img, isPrimary: false))
          .toList();

      setState(() {
        _details = details;
        _photos
          ..clear()
          ..addAll(loadedPhotos);
        _pendingDeleteStoredFileIds.clear();
      });
    } else {
      final msg =
          response.messageDetail ??
          'No package found for this tracking number.';
      MessageHelper.showIconSnackBar(context, message: msg, isSuccess: false);
      setState(() {
        _details = null;
        _photos.clear();
        _pendingDeleteStoredFileIds.clear();
      });
      _trackingFocusNode.requestFocus();
    }
  } catch (_) {
    if (!mounted) return;
    MessageHelper.showIconSnackBar(
      context,
      message: 'Unexpected error loading package details.',
      isSuccess: false,
    );
  }
}
```

## Data Models

### GetPackageTrackingDetailsResponse

The main response model containing all package information:

```dart
class GetPackageTrackingDetailsResponse {
  final List<PackageTrackingItemDetails> items;
  final bool allowUploadImages;
  final List<FileUrlResponse> images;
  final String? packageStatus;

  const GetPackageTrackingDetailsResponse({
    required this.items,
    required this.allowUploadImages,
    required this.images,
    this.packageStatus,
  });

  factory GetPackageTrackingDetailsResponse.fromJson(
    Map<String, dynamic> json,
  ) {
    final contentJson = json['content'] as Map<String, dynamic>? ?? {};

    final itemsJson = contentJson['items'] as List<dynamic>?;
    final items = itemsJson == null
        ? <PackageTrackingItemDetails>[]
        : itemsJson
            .whereType<Map<String, dynamic>>()
            .map(PackageTrackingItemDetails.fromJson)
            .toList();

    final images = <FileUrlResponse>[];

    final urlImagesJson = contentJson['urlImages'] as List<dynamic>?;
    if (urlImagesJson != null) {
      for (final item in urlImagesJson) {
        if (item is Map<String, dynamic>) {
          images.add(FileUrlResponse.fromJson(item));
        } else if (item is String) {
          images.add(FileUrlResponse.fromUrlString(item));
        }
      }
    }

    final allowUploadImages = contentJson['allowUploadImages'] as bool? ?? false;
    final packageStatus = contentJson['packageStatus'] as String?;

    return GetPackageTrackingDetailsResponse(
      items: items,
      allowUploadImages: allowUploadImages,
      images: images,
      packageStatus: packageStatus,
    );
  }
}
```

### PackageTrackingHistoryEntry

Individual tracking events:

```dart
class PackageTrackingHistoryEntry {
  final String? event;
  final String? remarks;
  final DateTime? dateTime;

  const PackageTrackingHistoryEntry({
    this.event,
    this.remarks,
    this.dateTime,
  });

  factory PackageTrackingHistoryEntry.fromJson(Map<String, dynamic> json) {
    final rawDate = json['dateTime'] as String?;
    DateTime? parsedDate;
    if (rawDate != null) {
      try {
        parsedDate = DateTime.tryParse(rawDate);
      } catch (_) {
        parsedDate = null;
      }
    }

    return PackageTrackingHistoryEntry(
      event: json['event'] as String?,
      remarks: json['remarks'] as String?,
      dateTime: parsedDate,
    );
  }
}
```

### Saving Photos

Upload new photos and delete removed ones:

```dart
Future<void> _saveImages() async {
  final packageId = _currentPackageId();
  final canUploadImages = _details?.allowUploadImages ?? false;

  if (packageId == null || !canUploadImages || _isSavingImages) return;
  final measurementService = context.read<MeasurementService>();

  setState(() {
    _isSavingImages = true;
  });

  try {
    final deleteIds = _pendingDeleteStoredFileIds.toList();

    final List<PackageImageEncodeFile> files = [];
    for (final item in _photos) {
      final remote = item.remoteFile;
      final file = item.localFile;

      if (remote != null && remote.storedFileId != 0) {
        files.add(PackageImageEncodeFile(storedFileId: remote.storedFileId));
      } else if (file != null) {
        final bytes = await File(file.path).readAsBytes();
        final encoded = base64Encode(bytes);
        final name = file.name;
        final ext = name.contains('.') ? name.split('.').last : 'jpg';

        files.add(
          PackageImageEncodeFile(
            identifier: file.path,
            encodeContent: encoded,
            fileExtension: ext,
            fileName: name,
          ),
        );
      }
    }

    if (deleteIds.isNotEmpty) {
      final deleteRequest = DeleteTrackingPackageImagesRequest(
        storedFileIds: deleteIds,
      );
      final deleteResponse = await measurementService
          .deleteTrackingPackageImages(deleteRequest);

      if (!mounted) return;

      if (!deleteResponse.isSuccessful) {
        final msg =
            deleteResponse.messageDetail ??
            'Unable to delete some images for this package.';
        MessageHelper.showIconSnackBar(
          context,
          message: msg,
          isSuccess: false,
        );
        return;
      }
    }

    if (files.isNotEmpty) {
      final registerRequest = RegisterPackageImagesByMobileRequest(
        packageId: packageId,
        files: files,
      );
      final registerResponse = await measurementService.registerPackageImages(
        registerRequest,
      );

      if (!mounted) return;

      if (!registerResponse.isSuccessful) {
        final msg =
            registerResponse.messageDetail ??
            'Unable to register images for this package.';
        MessageHelper.showIconSnackBar(
          context,
          message: msg,
          isSuccess: false,
        );
        return;
      }
    }

    if (!mounted) return;

    MessageHelper.showIconSnackBar(
      context,
      message: 'Package images updated successfully.',
      isSuccess: true,
    );
  } catch (_) {
    // Error handling
  }
}
```

### Image Reordering

Photos can be reordered via drag and drop:

```dart
void _reorderImages(int oldIndex, int newIndex) {
  setState(() {
    if (newIndex > oldIndex) {
      newIndex -= 1;
    }
    final item = _photos.removeAt(oldIndex);
    _photos.insert(newIndex, item);
  });
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Efficient Lookups">
    - Use debouncing for auto-search (80ms delay)
    - Cache recently viewed packages
    - Validate identifiers before API calls
    - Show loading indicators during fetch
  </Accordion>
  
  <Accordion title="Photo Management">
    - Limit to 5 images per package
    - Compress images before upload (85% quality, 1920x1080 max)
    - Allow reordering for better organization
    - Confirm before deletion
  </Accordion>
  
  <Accordion title="User Experience">
    - Auto-focus input after operations
    - Show clear status indicators
    - Display timestamps in local format
    - Handle empty states gracefully
  </Accordion>
</AccordionGroup>

<Tip>
  Use infrared scanners for faster tracking number input in high-volume environments. The system automatically triggers lookup after a brief delay.
</Tip>
